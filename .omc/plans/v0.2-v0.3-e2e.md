### Step 10 — E2E test (gated)

```python
import os
from pathlib import Path
import pytest
from roomestim.adapters.ace_challenge import list_rooms, load_room, dataset_name
from roomestim.reconstruct.materials import sabine_rt60, sabine_rt60_per_band

E2E_DIR_ENV = "ROOMESTIM_E2E_DATASET_DIR"

@pytest.mark.e2e
@pytest.mark.network
def test_e2e_rt60_characterisation(capsys):
    dataset_dir = os.environ.get(E2E_DIR_ENV)
    if not dataset_dir or not Path(dataset_dir).is_dir():
        pytest.skip(f"E2E test gated; set {E2E_DIR_ENV} to a populated ACE Challenge directory.")
    rooms = list_rooms(Path(dataset_dir))
    if not rooms:
        pytest.skip(f"{E2E_DIR_ENV} contains zero usable rooms.")
    errors_500hz = []
    errors_per_band = {b: [] for b in (125,250,500,1000,2000,4000)}
    for room_id in rooms:
        case = load_room(Path(dataset_dir), room_id)
        # Compute predicted RT60 from the synthesized RoomModel (geometry + material assignment)
        predicted_500hz = sabine_rt60(...)  # computed from case.room geometry & surfaces
        predicted_per_band = sabine_rt60_per_band(...)
        e500 = predicted_500hz - case.measured_rt60_500hz_s
        errors_500hz.append((room_id, e500))
        for band, predicted in predicted_per_band.items():
            measured = case.measured_rt60_per_band_s.get(band)
            if measured is not None:
                errors_per_band[band].append((room_id, predicted - measured))
        print(f"[{room_id}] 500Hz: predicted={predicted_500hz:.3f}s measured={case.measured_rt60_500hz_s:.3f}s err={e500:+.3f}s")
    # Write report
    write_report(dataset_name(), errors_500hz, errors_per_band)
    # Characterisation only — NO assert on magnitude.
```
